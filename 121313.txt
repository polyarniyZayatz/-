      return Math.max(1, discounted);
    }

    function calculateMailReward() {
      const streakPart = state.streak * state.streakBonus;
      const raw = (state.rewardPerMail + streakPart) * state.rewardMultiplier;
      return Math.max(1, Math.floor(raw));
    }

    function createMail() {
      const template = mailTemplates[Math.floor(Math.random() * mailTemplates.length)];
      state.mails.unshift({
        id: state.nextMailId++,
        from: template.from,
        subject: template.subject,
        body: template.body,
        hint: template.hint,
        reward: calculateMailReward()
      });
    }

    function openReply(mailId) {
      const mail = state.mails.find(m => m.id === mailId);
      if (!mail) return;

      state.currentReplyMail = mail;
      replyTitle.textContent = `Reply - ${mail.subject}`;
      replyMeta.innerHTML = `
        <div><strong>From:</strong> ${mail.from}</div>
        <div><strong>Subject:</strong> ${mail.subject}</div>
      `;
      originalMailBody.textContent = mail.body;
      replyTextarea.value = '';
      mistakeBox.textContent = 'Type the exact example text to complete the letter.';
      rewardPreview.textContent = `Correct letter reward: ${mail.reward} points`;
      renderTypingLayer();
      updateProgress();
      replyWindow.classList.remove('hidden');
      setStatus(`Opened reply editor for "${mail.subject}"`);
      replyTextarea.focus();
    }

    function closeReplyWindow() {
      replyWindow.classList.add('hidden');
      state.currentReplyMail = null;
      replyTextarea.value = '';
      editorDisplay.innerHTML = '';
      progressBox.textContent = '';
      mistakeBox.textContent = 'Type the exact example text to complete the letter.';
      rewardPreview.textContent = `Correct letter reward: ${calculateMailReward()} points`;
      setStatus('Reply window closed');
    }

    function renderTypingLayer() {
      const mail = state.currentReplyMail;
      if (!mail) {
        editorDisplay.innerHTML = '';
        return;
      }

      const typedValue = replyTextarea.value;
      const fullValue = mail.hint;
      const typedPart = escapeHtml(fullValue.slice(0, typedValue.length));
      const remainingPart = escapeHtml(fullValue.slice(typedValue.length));

      editorDisplay.innerHTML = `<span class="typed">${typedPart}</span><span class="remaining">${remainingPart}</span>`;
    }

    function updateProgress() {
      const mail = state.currentReplyMail;
      if (!mail) {
        progressBox.textContent = '';
        return;
      }
      progressBox.textContent = ` (${replyTextarea.value.length}/${mail.hint.length})`;
    }

    function completeMail(mail, isAuto = false) {
      state.points += mail.reward;
      state.totalCompletedMails += 1;
      state.streak += 1;
      if (state.streak > state.bestStreak) state.bestStreak = state.streak;
      if (isAuto) state.totalAutoReplies += 1;
      state.mails = state.mails.filter(m => m.id !== mail.id);

      if (state.currentReplyMail && state.currentReplyMail.id === mail.id) {
        closeReplyWindow();
      }

      setStatus(`${isAuto ? 'Auto reply sent' : 'Correct letter sent'}. Earned ${mail.reward} point(s)`);
      autoSave();
      renderAll();
    }

    function failCurrentLetter() {
      state.streak = 0;
      setStatus('Incorrect letter. No points awarded.');
      mistakeBox.textContent = 'Incorrect letter. Type the exact text to earn points.';
      autoSave();
      renderAll();
    }

    function sendReply() {
      const mail = state.currentReplyMail;
      if (!mail) return;

      if (replyTextarea.value !== mail.hint) {
        failCurrentLetter();
        return;
      }

      completeMail(mail, false);
    }

    function renderLoadingBlocks() {
      const totalBlocks = 22;
      loadingBar.innerHTML = '';
      for (let i = 0; i < totalBlocks; i++) {
        const block = document.createElement('div');
        block.className = 'win-progress-block';
        loadingBar.appendChild(block);
      }
    }

    function runAutoReplyAnimation(mail, done) {
      autoReplyBusy = true;
      loadingText.textContent = `Auto-answering: ${mail.subject}`;
      loadingSubtext.textContent = `Contact: ${mail.from}`;
      renderLoadingBlocks();
      loadingBar.style.width = '0%';
      loadingWindow.classList.remove('hidden');

      const duration = 900 + Math.floor(Math.random() * 500);
      const start = performance.now();

      function step(now) {
        const progress = Math.min(1, (now - start) / duration);
        loadingBar.style.width = `${progress * 100}%`;

        if (progress < 1) {
          requestAnimationFrame(step);
        } else {
          setTimeout(() => {
            loadingWindow.classList.add('hidden');
            autoReplyBusy = false;
            done();
            processAutoReplyQueue();
          }, 120);
        }
      }

      requestAnimationFrame(step);
    }

    function enqueueAutoReply(mailId) {
      if (!state.mails.find(m => m.id === mailId)) return;
      if (!autoReplyQueue.includes(mailId)) autoReplyQueue.push(mailId);
      processAutoReplyQueue();
    }

    function processAutoReplyQueue() {
      if (autoReplyBusy) return;
      if (!autoReplyQueue.length) return;

      const mailId = autoReplyQueue.shift();
      const mail = state.mails.find(m => m.id === mailId);
      if (!mail) {
        processAutoReplyQueue();
        return;
      }

      runAutoReplyAnimation(mail, () => {
        completeMail(mail, true);
      });
    }

    function autoReplyOne() {
      if (!state.mails.length) {
        setStatus('No e-mails available for auto reply');
        return;
      }
      enqueueAutoReply(state.mails[state.mails.length - 1].id);
    }

    function buyItem(id) {
      const item = shopItems.find(i => i.id === id);
      if (!item) return;

      const ownedCount = getItemOwnedCount(id);
      const cost = getDiscountedCost(item.baseCost, ownedCount);

      if (state.points < cost) {
        showPopup('Not Enough Points', `You need ${cost} points for ${item.name}.`);
        return;
      }

      state.points -= cost;
      item.effect();
      state.owned.push({
        id: item.id,
        name: item.name,
        description: item.description
      });

      setStatus(`Purchased ${item.name}`);
      autoSave();
      renderAll();
    }

    function renderStats() {
      pointsDisplay.textContent = state.points;
      perReplyDisplay.textContent = calculateMailReward();
      autoMailDisplay.textContent = state.autoMailRate;
      inboxCountDisplay.textContent = state.mails.length;
      streakDisplay.textContent = state.streak;
      autoReplyDisplay.textContent = state.autoReplyRate;
    }

    function renderMail() {
      if (!state.mails.length) {
        mailList.innerHTML = `<div class="mail-card">No incoming e-mails. Please wait for new mail...</div>`;
        return;
      }

      mailList.innerHTML = state.mails.map(mail => `
        <div class="mail-card">
          <div class="mail-subject">
            <span>${mail.subject}</span>
            <span class="mail-badge">+${mail.reward} pts</span>
          </div>
          <div class="mail-meta">From: ${mail.from}</div>
          <div class="mail-body">${mail.body}</div>
          <div class="mail-actions">
            <button class="action-btn" onclick="openReply(${mail.id})" type="button">Open</button>
            <button class="action-btn" onclick="queueMailAutoReply(${mail.id})" type="button">Auto Reply</button>
          </div>
        </div>
      `).join('');
    }

    function renderShop() {
      shopGrid.innerHTML = shopItems.map(item => {
        const ownedCount = getItemOwnedCount(item.id);
        const cost = getDiscountedCost(item.baseCost, ownedCount);

        return `
          <div class="shop-item">
            <div class="shop-icon">${item.icon}</div>
            <div class="shop-title">
              <span>${item.name}</span>
              <span>x${ownedCount}</span>
            </div>
            <div class="shop-desc">${item.description}</div>
            <div class="shop-cost">Cost: ${cost} points</div>
            <button class="buy-btn" onclick="buyItem('${item.id}')" type="button">Buy</button>
          </div>
        `;
      }).join('');
    }

    function renderOwned() {
      if (!state.owned.length) {
        ownedList.innerHTML = `<div class="owned-item">No bonuses owned yet.</div>`;
        return;
      }

      const grouped = {};
      state.owned.forEach(item => {
        if (!grouped[item.id]) {
          grouped[item.id] = {
            name: item.name,
            description: item.description,
            count: 0
          };
        }
        grouped[item.id].count += 1;
      });

      ownedList.innerHTML = Object.values(grouped).map(item => `
        <div class="owned-item">
          <div class="owned-item-title">${item.name} x${item.count}</div>
          <div>${item.description}</div>
        </div>
      `).join('');
    }

    function renderPanels() {
      document.getElementById('mailPanel').classList.toggle('hidden', state.activePanel !== 'mail');
      document.getElementById('shopPanel').classList.toggle('hidden', state.activePanel !== 'shop');
      document.getElementById('upgradesPanel').classList.toggle('hidden', state.activePanel !== 'upgrades');
      document.getElementById('helpPanel').classList.toggle('hidden', state.activePanel !== 'help');

      document.querySelectorAll('.nav-item').forEach(item => {
        item.classList.toggle('active', item.dataset.panel === state.activePanel);
      });
    }

    function renderAll() {
      renderStats();
      renderMail();
      renderShop();
      renderOwned();
      renderPanels();
    }

    function saveGame() {
      const toSave = {
        ...state,
        currentReplyMail: null
      };
      localStorage.setItem(SAVE_KEY, JSON.stringify(toSave));
      setSaveState('Save: stored');
    }

    function autoSave() {
      saveGame();
    }

    function loadGame() {
      const raw = localStorage.getItem(SAVE_KEY);
      if (!raw) return false;

      try {
        const parsed = JSON.parse(raw);
        state = {
          ...structuredClone(defaultState),
          ...parsed,
          currentReplyMail: null
        };
        return true;
      } catch (error) {
        console.error('Save load error:', error);
        return false;
      }
    }

    function resetGame() {
      localStorage.removeItem(SAVE_KEY);
      state = structuredClone(defaultState);
      autoReplyQueue = [];
      autoReplyBusy = false;
      loadingWindow.classList.add('hidden');
      for (let i = 0; i < 3; i++) createMail();
      closeReplyWindow();
      renderAll();
      setStatus('Save reset completed');
      setSaveState('Save: reset');
    }

    replyTextarea.addEventListener('keydown', (e) => {
      const mail = state.currentReplyMail;
      if (!mail) return;

      const allowedControlKeys = ['Backspace', 'Delete', 'ArrowLeft', 'ArrowRight', 'ArrowUp', 'ArrowDown', 'Tab', 'Home', 'End'];
      if (allowedControlKeys.includes(e.key)) return;

      if (e.ctrlKey || e.metaKey || e.altKey) {
        e.preventDefault();
        return;
      }

      if (e.key.length !== 1) return;

      const current = replyTextarea.value;
      const expectedChar = mail.hint[current.length];

      if (e.key !== expectedChar) {
        if (state.keyboardHelpLevel > 0) {
          e.preventDefault();
          replyTextarea.value += expectedChar;
          renderTypingLayer();
          updateProgress();
          mistakeBox.textContent = `Auto-correct helped: expected "${expectedChar}"`;
          return;
        }

        e.preventDefault();
        mistakeBox.textContent = `Wrong key blocked. Expected "${expectedChar}"`;
        setStatus('Wrong character blocked');
      } else {
        mistakeBox.textContent = 'Good. Keep typing the correct reply.';
      }
    });

    replyTextarea.addEventListener('paste', (e) => {
      e.preventDefault();
      mistakeBox.textContent = 'Paste is disabled. Type the letter manually.';
    });

    replyTextarea.addEventListener('input', () => {
      const mail = state.currentReplyMail;
      if (!mail) return;

      if (!mail.hint.startsWith(replyTextarea.value)) {
        replyTextarea.value = mail.hint.slice(0, replyTextarea.value.length - 1);
      }

      renderTypingLayer();
      updateProgress();

      if (replyTextarea.value === mail.hint) {
        mistakeBox.textContent = 'Letter complete. You can now send it.';
      }
    });

    document.querySelectorAll('.nav-item').forEach(item => {
      item.addEventListener('click', () => {
        state.activePanel = item.dataset.panel;
        setStatus(`Opened ${item.textContent}`);
        renderPanels();
      });
    });

    document.querySelectorAll('.menu-bar span').forEach(span => {
      span.addEventListener('click', () => {
        setStatus(`${span.textContent} menu is not implemented yet`);
      });
    });

    document.getElementById('closeReplyBtn').addEventListener('click', closeReplyWindow);
    document.getElementById('cancelReplyBtn').addEventListener('click', closeReplyWindow);
    document.getElementById('sendReplyBtn').addEventListener('click', sendReply);
    document.getElementById('popupCloseBtn').addEventListener('click', closePopup);
    document.getElementById('popupOkBtn').addEventListener('click', closePopup);
    document.getElementById('saveBtn').addEventListener('click', () => {
      saveGame();
      setStatus('Game saved manually');
    });
    document.getElementById('resetBtn').addEventListener('click', resetGame);
    document.getElementById('spawnMailBtn').addEventListener('click', () => {
      createMail();
      renderAll();
      autoSave();
      setStatus('Manual mail generated');
    });
    document.getElementById('autoReplyOneBtn').addEventListener('click', autoReplyOne);

    if (!loadGame()) {
      for (let i = 0; i < 3; i++) createMail();
      setSaveState('Save: new game');
    } else {
      setStatus('Save loaded successfully');
      setSaveState('Save: loaded');
    }

    setInterval(() => {
      for (let i = 0; i < state.autoMailRate; i++) createMail();
      renderAll();
      autoSave();
      setStatus('New e-mail(s) arrived');
    }, 6000);

    setInterval(() => {
      if (!state.mails.length || state.autoReplyRate <= 0) return;
      const repliesToQueue = Math.min(state.autoReplyRate, state.mails.length);
      for (let i = 0; i < repliesToQueue; i++) {
        const target = state.mails[state.mails.length - 1 - i];
        if (target) enqueueAutoReply(target.id);
      }
    }, 12000);

    setInterval(() => {
      autoSave();
    }, 15000);

    renderAll();

    window.openReply = openReply;
    window.buyItem = buyItem;
    window.queueMailAutoReply = enqueueAutoReply;
  </script>
